import Vue from 'vue'

// only import the icons you use to reduce bundle size import 'vue-awesome/icons/flag'

/ Register component with one of 2 methods /

import Icon from 'vue-awesome/components/Icon'

// globally (in your main .js file) Vue.component('icon', Icon)

// or locally (in your component file) / export default { components: { Icon } }/
